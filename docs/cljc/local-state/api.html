(defn update-state!
  [state-id f & params]
  (letfn [(f0 [%] (apply f % params))]
         (swap! state/STATES update state-id f0)))

(defn assoc-state!
  [state-id & ksnv]
  (swap! state/STATES assoc-in (-> ksnv (vector/remove-last-item)
                                        (vector/cons-item state-id))
                               (-> ksnv (vector/last-item))))
