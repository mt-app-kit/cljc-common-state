(defn get-cursor
  [state-id]
  (if-let [cursor (-> state/CURSORS deref (get state-id))]
          (-> cursor)
          (let [cursor (utils/cursor state/COMMON-STATE [state-id])]
               (->> cursor (swap! state/CURSORS assoc state-id))
               (->> cursor))))

(defn get-state
  [state-id & keys]
  (let [cursor (get-cursor state-id)]
       (-> cursor deref (get-in keys))))

(defn update-state!
  [state-id f & params]
  (letfn [(f0 [%] (apply f % params))]
         (swap! state/COMMON-STATE update state-id f0)))

(defn assoc-state!
  [state-id & ksnv]
  (swap! state/COMMON-STATE assoc-in (-> ksnv (vector/remove-last-item)
                                              (vector/cons-item state-id))
                                     (-> ksnv (vector/last-item))))
